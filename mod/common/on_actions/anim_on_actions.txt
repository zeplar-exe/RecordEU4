    log = "ANIMATED_REPLAY on_startup||[Root.GetTag]"

    if = {
        limit = {
            NOT = { has_global_flag = read_prov_id }

        set_prov_vars = yes

        every_province = {
            log = "ANIMATED_REPLAY startup_province||[This.prov_id.GetValue]||[This.GetName]"
        }

        log = "ANIMATED_REPLAY startup_province_ownership||[This.prov_id.GetValue]||[Prev.GetTag]"
    }

    country_event = {
        id = anim_util_events.1
        days = 1
    }
}
