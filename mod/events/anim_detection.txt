namespace = anim_detection_events

# detect province stuff
province_event = {
    id = anim_detection_events.1
    hidden = yes

    trigger = {
        has_siege = yes
    }

    mean_time_to_happen = {
        days = 1
    }

    option = {
        ai_chance = 1
    }

    every_country = {
        limit = {
            Root = {
                sieged_by = Prev
            }
        }

        export_to_variable = {
            which = anim_garrison
            value = trigger_value:garrison
        }

        log = "ANIMATED_REPLAY on_province_siege_progress||[Root.prov_id.GetValue]||[This.anim_garrison.GetValue]||[This.GetTag]"
    }
}
